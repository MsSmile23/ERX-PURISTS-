				scope = ERB  
				days = 180
    		}
		}
	}

	option = { 
		name = purists_ead_military.3.b
		custom_effect_tooltip = purists_ead_military.3.b.tt
		GRD = {
			add_dynamic_modifier = {
				modifier = Purists_sigma_subotaje_active
				scope = ERB  
				days = 180
    		}
		}
	}

	option = { 
		name = purists_ead_military.3.c
		custom_effect_tooltip = purists_ead_military.3.c.tt
		custom_effect_tooltip = purists_extra_weapons.tt
		trigger = {
			has_equipment = {
				support_equipment > 200
			}

			has_equipment = {
				infantry_equipment > 1000
			}
		}
		DES = {
			add_timed_idea = {
				idea = Purists_mercenary_support
				days = 180
			}
			add_equipment_to_stockpile = {
				type = infantry_equipment
				amount = 2000
				producer = ERB
			}
			add_equipment_to_stockpile = {
				type = support_equipment
				amount = 400
				producer = ERB
			}
		}
		GRD = {
			add_dynamic_modifier = {
				modifier = Purists_sigma_subotaje_active
				scope = ERB  
				days = 180
    		}
		}

		ERB = {
			add_equipment_to_stockpile = {
				type = infantry_equipment
				amount = -1000
			}
			add_equipment_to_stockpile = {
				type = support_equipment
				amount = -200
			}
		}
	}

	option = { 
		name = purists_ead_military.3.e
		add_political_power = 50
	}
}
